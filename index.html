구글 시트 열기

2

코드 교체

기존 내용 전체 삭제 후 아래 코드 붙여넣기

function doGet(e) { try { const callback = e.parameter.callback; const action = e.parameter.action; if (action === 'update') return handleUpdate(e, callback); const sheet = SpreadsheetApp.getActiveSpreadsheet() .getSheetByName('🗓️ 전체 일정'); if (!sheet) { const j = JSON.stringify({ok:false,error:'시트없음'}); return out(callback?(callback+'('+j+')'):j, callback?'js':'json'); } const rows = sheet.getDataRange().getValues(); const data = []; for (let i = 0; i < rows.length; i++) { const row = rows[i]; if (!row[0]) continue; let date = ''; try { date = row[0] instanceof Date ? Utilities.formatDate(row[0],'Asia/Seoul','yyyy-MM-dd') : String(row[0]); } catch(err) { date = String(row[0]); } if (!date.match(/\d{4}-\d{2}-\d{2}/)) continue; data.push({row:i, data:[date,String(row[1]||''),String(row[2]||''),String(row[3]||'-'),String(row[4]||'-'),String(row[5]||'-'),String(row[6]||'-'),String(row[7]||'')]}); } const j = JSON.stringify({ok:true, data:data}); return out(callback?(callback+'('+j+')'):j, callback?'js':'json'); } catch(err) { const j = JSON.stringify({ok:false,error:err.message}); const cb = e.parameter?e.parameter.callback:null; return out(cb?(cb+'('+j+')'):j, cb?'js':'json'); } } function handleUpdate(e, callback) { try { const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName('🗓️ 전체 일정'); const rowIndex = parseInt(e.parameter.row) + 1; sheet.getRange(rowIndex,4).setValue(e.parameter.driver||'-'); sheet.getRange(rowIndex,5).setValue(e.parameter.p1||'-'); sheet.getRange(rowIndex,6).setValue(e.parameter.p2||'-'); sheet.getRange(rowIndex,7).setValue(e.parameter.p3||'-'); sheet.getRange(rowIndex,8).setValue(e.parameter.note||''); const j = JSON.stringify({ok:true}); return out(callback?(callback+'('+j+')'):j, callback?'js':'json'); } catch(err) { const j = JSON.stringify({ok:false,error:err.message}); return out(callback?(callback+'('+j+')'):j, callback?'js':'json'); } } function out(text, type) { return ContentService.createTextOutput(text) .setMimeType(type==='js'?ContentService.MimeType.JAVASCRIPT:ContentService.MimeType.JSON); }
